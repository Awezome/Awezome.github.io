            if($data[$j]<$data[$min]){
                $min=$j;
            }
        }
        self::exch($data,$i,$min);

static function quick(array &$data,$lo,$hi){
    if($hi<=$lo){
        return ;
    }

    //partition
    $i=$lo;
    $j=$hi+1;
    $v=$data[$lo];
    while(1){
        while($data[++$i]<$v){
            if($i==$hi){
                break;
            }
        }
        while($v<$data[--$j]){
            if($j==$lo){
                break;
            }
        }
        if($i>=$j){
            break;
        }
        self::exch($data,$i,$j);
    }
    self::each($data,$lo,$j);
    $k=$j;

    //
    quick($data,$lo,$k-1);
    quick($data,$k+1,$hi);
}

`static function exch(&$data,$i,$j){
    $tmp=$data[$i];
    $data[$i]=$data[$j];
    $data[$j]=$tmp;
}
